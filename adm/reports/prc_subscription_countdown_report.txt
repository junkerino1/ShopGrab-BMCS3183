CREATE OR REPLACE PROCEDURE prc_sub_countdown_report AS

    CURSOR cur_countdown IS
        SELECT
            customer_id,
            name,
            email,
            days_left,
            end_date
        FROM vw_subscription_countdown
        ORDER BY days_left ASC, name ASC;

    v_total_count         NUMBER := 0;
    v_today_count         NUMBER := 0;
    v_soon_count          NUMBER := 0;
    v_week_count          NUMBER := 0;
    v_has_data            BOOLEAN := FALSE;

BEGIN
    DBMS_OUTPUT.PUT_LINE(RPAD('=', 110, '='));
    DBMS_OUTPUT.PUT_LINE(LPAD('SUBSCRIPTION COUNTDOWN REPORT (NEXT 7 DAYS)', 70));
    DBMS_OUTPUT.PUT_LINE('REPORT DATE: ' || TO_CHAR(SYSDATE, 'DD-MON-YYYY HH24:MI'));
    DBMS_OUTPUT.PUT_LINE(RPAD('=', 110, '='));

    DBMS_OUTPUT.PUT_LINE(
        RPAD('CUS ID', 8) || ' | ' ||
        RPAD('CUS NAME', 14) || ' | ' ||
        RPAD('EMAIL', 24) || ' | ' ||
        LPAD('DAYS LEFT', 10) || ' | ' ||
        RPAD('END DATE', 12) || ' | ' ||
        RPAD('STATUS', 14)
    );
    DBMS_OUTPUT.PUT_LINE(RPAD('-', 110, '-'));

    FOR rec IN cur_countdown LOOP
        v_has_data := TRUE;
        v_total_count := v_total_count + 1;

        IF rec.days_left = 0 THEN
            v_today_count := v_today_count + 1;
        ELSIF rec.days_left BETWEEN 1 AND 3 THEN
            v_soon_count := v_soon_count + 1;
        ELSE
            v_week_count := v_week_count + 1;
        END IF;

        DBMS_OUTPUT.PUT_LINE(
            RPAD(rec.customer_id, 8) || ' | ' ||
            RPAD(INITCAP(rec.name), 14) || ' | ' ||
            RPAD(LOWER(rec.email), 24) || ' | ' ||
            LPAD(rec.days_left, 10) || ' | ' ||
            RPAD(TO_CHAR(rec.end_date, 'DD-MM-YYYY'), 12) || ' | ' ||
            RPAD(
                CASE
                    WHEN rec.days_left = 0 THEN 'Expires Today'
                    WHEN rec.days_left <= 3 THEN 'Expiring Soon'
                    ELSE 'Active'
                END,
                14
            )
        );
    END LOOP;

    DBMS_OUTPUT.PUT_LINE(RPAD('-', 110, '-'));

    IF v_has_data THEN
        DBMS_OUTPUT.PUT_LINE('TOTAL RECORDS           : ' || v_total_count);
        DBMS_OUTPUT.PUT_LINE('EXPIRES TODAY           : ' || v_today_count);
        DBMS_OUTPUT.PUT_LINE('EXPIRING IN 1-3 DAYS    : ' || v_soon_count);
        DBMS_OUTPUT.PUT_LINE('EXPIRING IN 4-7 DAYS    : ' || v_week_count);
    ELSE
        DBMS_OUTPUT.PUT_LINE('No active subscriptions are expiring within the next 7 days.');
    END IF;

    DBMS_OUTPUT.PUT_LINE(RPAD('=', 110, '='));

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END prc_sub_countdown_report;
/
