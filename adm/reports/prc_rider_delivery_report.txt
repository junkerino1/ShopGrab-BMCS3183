CREATE OR REPLACE PROCEDURE prc_rider_delivery_report AS

    -- 1. Outer Cursor: Loop through distinct months present in completed orders
    CURSOR cur_months IS
        SELECT DISTINCT TO_CHAR(order_datetime, 'YYYY-MM') AS month_str
        FROM CUST_ORDER
        WHERE order_status = 'Completed' 
          AND order_type = 'D'
        ORDER BY month_str;

    -- 2. Middle Cursor: Loop through active platforms
    CURSOR cur_platforms IS
        SELECT platform_id, platform_name
        FROM PLATFORM
        ORDER BY platform_name;

    -- 3. Inner Cursor: Get total deliveries per rider for the specific month and platform
    CURSOR cur_rider_deliveries (p_month VARCHAR2, p_platform_id NUMBER) IS
        SELECT 
            r.rider_id,
            r.name AS rider_name,
            COUNT(o.order_id) AS total_deliveries
        FROM RIDER r
        JOIN CUST_ORDER o ON r.rider_id = o.rider_id
        WHERE o.order_status = 'Completed'
          AND o.order_type = 'D'
          AND r.platform_id = p_platform_id
          AND TO_CHAR(o.order_datetime, 'YYYY-MM') = p_month
        GROUP BY r.rider_id, r.name
        ORDER BY total_deliveries DESC, r.name;

    v_month_rec     cur_months%ROWTYPE;
    v_plat_rec      cur_platforms%ROWTYPE;
    v_rider_rec     cur_rider_deliveries%ROWTYPE;

    v_plat_deliveries NUMBER;
    v_month_deliveries NUMBER;
    v_has_data BOOLEAN;

BEGIN
    DBMS_OUTPUT.PUT_LINE(LPAD('=', 80, '='));
    DBMS_OUTPUT.PUT_LINE('MONTHLY PLATFORM RIDER DELIVERY REPORT');
    DBMS_OUTPUT.PUT_LINE(LPAD('=', 80, '='));

    OPEN cur_months;
    LOOP
        FETCH cur_months INTO v_month_rec;
        EXIT WHEN cur_months%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE(CHR(10));
        DBMS_OUTPUT.PUT_LINE('MONTH: ' || v_month_rec.month_str);
        DBMS_OUTPUT.PUT_LINE(LPAD('-', 80, '-'));

        v_month_deliveries := 0;

        OPEN cur_platforms;
        LOOP
            FETCH cur_platforms INTO v_plat_rec;
            EXIT WHEN cur_platforms%NOTFOUND;

            -- We check if this platform actually has riders before printing its header
            v_plat_deliveries := 0;
            v_has_data := FALSE;

            OPEN cur_rider_deliveries(v_month_rec.month_str, v_plat_rec.platform_id);
            LOOP
                FETCH cur_rider_deliveries INTO v_rider_rec;
                EXIT WHEN cur_rider_deliveries%NOTFOUND;

                IF NOT v_has_data THEN
                    DBMS_OUTPUT.PUT_LINE('PLATFORM: ' || v_plat_rec.platform_name);
                    DBMS_OUTPUT.PUT_LINE(
                        RPAD('RIDER ID',       10, ' ') || ' | ' ||
                        RPAD('RIDER NAME',     35, ' ') || ' | ' ||
                        LPAD('DELIVERIES',     25, ' ')
                    );
                    DBMS_OUTPUT.PUT_LINE(LPAD('-', 80, '-'));
                    v_has_data := TRUE;
                END IF;

                DBMS_OUTPUT.PUT_LINE(
                    RPAD(v_rider_rec.rider_id,             10, ' ') || ' | ' ||
                    RPAD(v_rider_rec.rider_name,           35, ' ') || ' | ' ||
                    LPAD(v_rider_rec.total_deliveries,     25, ' ')
                );

                v_plat_deliveries := v_plat_deliveries + v_rider_rec.total_deliveries;
            END LOOP;
            CLOSE cur_rider_deliveries;

            -- Print Platform Total if data exists
            IF v_has_data THEN
                DBMS_OUTPUT.PUT_LINE(LPAD('-', 80, '-'));
                DBMS_OUTPUT.PUT_LINE(
                    RPAD('TOTAL FOR ' || v_plat_rec.platform_name, 48, ' ') || ' | ' ||
                    LPAD(v_plat_deliveries, 25, ' ')
                );
                DBMS_OUTPUT.PUT_LINE(' ');

                v_month_deliveries := v_month_deliveries + v_plat_deliveries;
                DBMS_OUTPUT.PUT_LINE(LPAD('-', 80, '-'));
            END IF;

        END LOOP;
        CLOSE cur_platforms;

        -- Print Monthly Total
        
        IF v_month_deliveries > 0 THEN
            DBMS_OUTPUT.PUT_LINE(
                RPAD('TOTAL DELIVERIES IN ' || v_month_rec.month_str, 48, ' ') || ' | ' ||
                LPAD(v_month_deliveries, 25, ' ')
            );
        ELSE
            DBMS_OUTPUT.PUT_LINE('No deliveries completely executed in this month.');
        END IF;

    END LOOP;
    CLOSE cur_months;

END prc_rider_delivery_report;
/
