CREATE OR REPLACE PROCEDURE prc_customer_retention_report AS

    -- Outer cursor: customers with coin balance above threshold.
    CURSOR cur_customers IS
        SELECT
            c.customer_id,
            c.username,
            cw.total_coins
        FROM CUSTOMER c
        JOIN COIN_WALLET cw ON c.customer_id = cw.customer_id
        WHERE cw.total_coins > 100
        ORDER BY cw.total_coins DESC, c.username ASC;

    -- Inner cursor: get latest order date and inactivity days for one customer.
    CURSOR cur_last_order (p_customer_id CUSTOMER.customer_id%TYPE) IS
        SELECT
            MAX(o.order_datetime) AS last_order_date,
            TRUNC(SYSDATE) - TRUNC(MAX(o.order_datetime)) AS days_since_last_order
        FROM CUST_ORDER o
        WHERE o.customer_id = p_customer_id;

    v_last_order_date   DATE;
    v_days_inactive     NUMBER;

    v_has_data BOOLEAN := FALSE;
    v_total_customers NUMBER := 0;

BEGIN
    DBMS_OUTPUT.PUT_LINE(RPAD('=', 60, '='));
    DBMS_OUTPUT.PUT_LINE(LPAD('CUSTOMER RETENTION REPORT', 52));
    DBMS_OUTPUT.PUT_LINE('(Coins > 100 and no orders in last 7 days)');
    DBMS_OUTPUT.PUT_LINE(RPAD('=', 60, '='));

    DBMS_OUTPUT.PUT_LINE(
        RPAD('CUS ID', 8) || ' | ' ||
        RPAD('USERNAME', 14) || ' | ' ||
        LPAD('COINS', 8) || ' | ' ||
        RPAD('LAST ORDER', 10) || ' | ' ||
        LPAD('INACTIVE', 8)
    );
    DBMS_OUTPUT.PUT_LINE(RPAD('-', 60, '-'));

    FOR rec_c IN cur_customers LOOP
        OPEN cur_last_order(rec_c.customer_id);
        FETCH cur_last_order INTO v_last_order_date, v_days_inactive;
        CLOSE cur_last_order;

        IF v_last_order_date IS NULL OR v_days_inactive >= 7 THEN
            v_has_data := TRUE;
            v_total_customers := v_total_customers + 1;

            DBMS_OUTPUT.PUT_LINE(
                RPAD(rec_c.customer_id, 8) || ' | ' ||
                RPAD(rec_c.username, 14) || ' | ' ||
                LPAD(rec_c.total_coins, 8) || ' | ' ||
                RPAD(NVL(TO_CHAR(v_last_order_date, 'DD-MM-YYYY'), 'Never'), 10) || ' | ' ||
                LPAD(NVL(TO_CHAR(v_days_inactive), '-'), 8)
            );
        END IF;
    END LOOP;

    DBMS_OUTPUT.PUT_LINE(RPAD('-', 60, '-'));

    IF NOT v_has_data THEN
        DBMS_OUTPUT.PUT_LINE('No customers meet the retention criteria.');
    ELSE
        DBMS_OUTPUT.PUT_LINE(RPAD('=', 60, '='));
        DBMS_OUTPUT.PUT_LINE('TOTAL CUSTOMERS AT CHURN RISK : ' || v_total_customers);
        DBMS_OUTPUT.PUT_LINE(RPAD('=', 60, '='));
    END IF;

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END prc_customer_retention_report;
