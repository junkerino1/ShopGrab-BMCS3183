CREATE OR REPLACE PROCEDURE prc_add_to_cart (
    p_item_id       IN NUMBER,
    p_restaurant_id IN NUMBER,
    p_quantity      IN NUMBER,
    p_remark        IN VARCHAR2 DEFAULT NULL
) AS
    v_cart_restaurant_id  NUMBER;
    v_item_name           MENU_ITEM.name%TYPE;
    v_unit_price          MENU_ITEM.price%TYPE;
    v_subtotal            NUMBER(8,2);
    v_existing_qty        NUMBER;
    v_new_cart_id         NUMBER;
BEGIN
    -- Business validation: quantity must be positive.
    IF p_quantity IS NULL OR p_quantity <= 0 THEN
        RAISE_APPLICATION_ERROR(-20012, 'Invalid quantity. Quantity must be greater than 0.');
    END IF;

    -- Business validation: cart can only contain items from one restaurant.
    BEGIN
        SELECT restaurant_id
        INTO v_cart_restaurant_id
        FROM TEMP_CART
        WHERE ROWNUM = 1;

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            NULL; -- Empty cart, so restaurant constraint is not violated.
    END;

    -- Validate: item must exist and be available in the given restaurant
    BEGIN
        SELECT name, price
        INTO v_item_name, v_unit_price
        FROM MENU_ITEM
        WHERE item_id = p_item_id
          AND restaurant_id = p_restaurant_id
          AND is_available = 'Y';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RAISE_APPLICATION_ERROR(-20011, 'Item not found or unavailable for the selected restaurant.');
    END;

    -- If item already exists in cart, just update the quantity
    BEGIN
        SELECT quantity INTO v_existing_qty
        FROM TEMP_CART
        WHERE item_id = p_item_id;

        v_subtotal := v_unit_price * (v_existing_qty + p_quantity);

        UPDATE TEMP_CART
        SET quantity = v_existing_qty + p_quantity,
            subtotal = v_subtotal,
            remark   = NVL(p_remark, remark)
        WHERE item_id = p_item_id;

        DBMS_OUTPUT.PUT_LINE('------------------------------------------');
        DBMS_OUTPUT.PUT_LINE('Cart updated: ' || v_item_name);
        DBMS_OUTPUT.PUT_LINE('New Quantity : ' || (v_existing_qty + p_quantity));
        DBMS_OUTPUT.PUT_LINE('New Subtotal : RM ' || TO_CHAR(v_subtotal, 'FM9999.00'));
        DBMS_OUTPUT.PUT_LINE('------------------------------------------');
        RETURN;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            NULL; -- Item not yet in cart, proceed to insert
    END;

    -- Insert as new cart row
    v_subtotal := v_unit_price * p_quantity;

    SELECT NVL(MAX(cart_item_id), 0) + 1
    INTO v_new_cart_id
    FROM TEMP_CART;

    INSERT INTO TEMP_CART (cart_item_id, item_id, restaurant_id, item_name, unit_price, quantity, subtotal, remark)
    VALUES (v_new_cart_id, p_item_id, p_restaurant_id, v_item_name, v_unit_price, p_quantity, v_subtotal, p_remark);

    DBMS_OUTPUT.PUT_LINE('------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Added to cart : ' || v_item_name);
    DBMS_OUTPUT.PUT_LINE('Quantity      : ' || p_quantity);
    DBMS_OUTPUT.PUT_LINE('Subtotal      : RM ' || TO_CHAR(v_subtotal, 'FM9999.00'));
    IF p_remark IS NOT NULL THEN
        DBMS_OUTPUT.PUT_LINE('Remark        : ' || p_remark);
    END IF;
    DBMS_OUTPUT.PUT_LINE('------------------------------------------');

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;
/
