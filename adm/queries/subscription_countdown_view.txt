CREATE OR REPLACE VIEW vw_subscription_countdown AS
SELECT
    c.customer_id AS customer_id,
    c.username AS name,
    c.email AS email,
    s.end_date,
    CEIL(s.end_date - SYSDATE) AS days_left
FROM CUSTOMER c
JOIN SUBSCRIPTION s ON c.customer_id = s.customer_id
WHERE s.status = 'Active'
  AND s.end_date BETWEEN SYSDATE AND SYSDATE + 7;
